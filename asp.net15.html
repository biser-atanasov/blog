  1. Admin Dropdown Menu

If the user is admin, we want him to see additional button in the navbar:

We can do this by adding a conditional statement in the "Views/Shared/_LoginPartial.cshtml":

Paste this code inside the if statement:

<li class="dropdown">

<a href="#" class="dropdown-toggle" data-toggle="dropdown" role="button" aria-expanded="false">Admin<span class="caret"></span></a>

<ul class="dropdown-menu" role="menu">

<li>@Html.ActionLink("Users", "List", "User")</li>

</ul>

</li>
This will create the dropdown menu. If you are logged as admin, you should see it.

Inside the new controller create a new action name called List, which will get all users from the database and pass them to the view.

Make the Index action to redirect to List action:

  1. List Users View

Create a new view inside "Views/User" called List and paste the following code:

@model List<Blog.Models.ApplicationUser>


@{

ViewBag.Title = "List";

}


<div class="container">

<div class="well">

<h2>Users</h2>


<table class="table table-striped table-hover">

<thead>

<tr>

<th>ID</th>

<th>Email</th>

<th>Actions</th>

</tr>

</thead>


<tbody>

@foreach (var user in Model)

{

<tr>

<td>@user.Id</td>

<td>@user.Email</td>

<td>

@Html.ActionLink("Edit", "Edit", "User", new { @id = user.Id }, null)

@Html.ActionLink("Delete", "Delete", "User", new { @id = user.Id }, null)

</td>

</tr>

}

</tbody>

</table>

</div>

</div>


In this view, we are just listing users with some information about them and action links for editing and deleting users.
